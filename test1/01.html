第一种的实现方式

var box = $('box');

$('larger').onclick = function(){

box.style.width = "300px";

}

$('heighter').onclick = function(){

box.style.height = "300px";

}

$('white').onclick = function(){

box.style.backgroundColor = "white";

}

$('hide').onclick = function(){

box.style.display = "none";

}

$('reset').onclick = function(){

box.style.display = "block";

box.style.width = "200px";

box.style.height = "200px";

box.style.backgroundColor = "#000";

}

第二种实现方式

function changeStyle(elem,attr,value){

elem.style[attr] = value;

}

window.onload = function ()

{

var oBtn = document.getElementsByTagName("input");

var oDiv = $("box2");

var oAtt = ["width","height","background","display","display"];

var oVal = ["300px","300px","red","none","block"];

for (var i = 0; i < oBtn.length; i++)

{

oBtn[i].index = i;

oBtn[i].onclick = function ()

{

console.log(this.index,oBtn.length-1,oDiv.style.cssText)

//这里需要判断index，如果是最后一个重置按钮的话，新设置的div的cssText设置为""

this.index == oBtn.length - 1 && (oDiv.style.cssText = ""); //== > &&

changeStyle(oDiv, oAtt[this.index], oVal[this.index])

}

}

};
