4.关于编码

5.截取字符串

		//slice(start,last);
		var str = "hanyan";
		str.slice(0,5); //hanya
		str.slice(1,1); //""

		//substr(start,length)
		var str = "hanyan"
		str.substr(0,5); //hanya
		str.substr(1,1) //a

		//substring(start,stop) 不接受负值
		var str = "hanyan";
		str.substring(0,1) //h
