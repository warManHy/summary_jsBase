2.鼠标点击事件 
function mouseWhichBtn(event){
	var rightBox = document.getElementsByClassName("rightBox")[0];
	var btnNum = event.button;
	if(btnNum==0){
		//左键
		alert("您点击了左键！");
		rightBox.style.display="none";
	}
	if(btnNum==1){
		//滚轮
		alert("您点击了滚轮！");
	}
	if(btnNum==2){
		//阻止默认事件
		document.getElementById("div1").oncontextmenu =function(){
			return false;
		};
		//右键
		var offsetL = document.getElementById("div1").offsetLeft;
		var offsetT = document.getElementById("div1").offsetTop;

		var rightBoxLeft = event.clientX - offsetL + "px";
		var rightBoxTop = event.clientY - offsetT + "px";
		console.log(rightBoxLeft,rightBoxTop);
		rightBox.style.left = rightBoxLeft;
		rightBox.style.top = rightBoxTop;
		rightBox.style.display = "block";

	}
}
